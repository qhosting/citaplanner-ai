# [AURUM PROTOCOL] - DOCKERFILE PRODUCTION MIRROR
# PRODUCTO: CITAPLANNER ELITE BUSINESS SUITE (SaaS EDITION)
# REVISIÓN: 2026.01.14 (Build Fix - Path Dependency Removal)
# STATUS: PRODUCTION DEPLOYMENT READY

# --- FASE 1: BUILDER ---
FROM node:20-alpine AS builder

# Dependencias de sistema necesarias para binarios nativos
RUN apk add --no-cache libc6-compat python3 make g++

WORKDIR /app

# Argumentos de construcción inyectados por el plano de control
ARG DATABASE_URL
ARG API_KEY
ARG NEXTAUTH_SECRET
ARG DOMAIN_URL
ARG REDIS_URL

# Variables de entorno para el proceso de compilación
ENV DATABASE_URL=$DATABASE_URL
ENV API_KEY=$API_KEY
ENV GEMINI_API_KEY=$API_KEY
ENV NEXTAUTH_SECRET=$NEXTAUTH_SECRET
ENV DOMAIN_URL=$DOMAIN_URL
ENV REDIS_URL=$REDIS_URL

# Copiamos la definición de paquetes corregida
COPY package.json ./

# Instalación de dependencias (legacy-peer-deps para compatibilidad React 19)
RUN npm install --legacy-peer-deps

# Inyección del código fuente tras la instalación
COPY . .

# Generación del bundle de producción (Vite + Rollup)
RUN npm run build

# Purga de dependencias de desarrollo para el runtime
RUN npm prune --production

# --- FASE 2: RUNNER ---
FROM node:20-alpine AS runner

WORKDIR /app

ENV NODE_ENV=production
ENV PORT=3000

# Seguridad: Usuario no privilegiado para ejecución
RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 citaplanner

# Preparación de volúmenes de persistencia
RUN mkdir -p /app/uploads && chown -R citaplanner:nodejs /app/uploads

# Copia selectiva de artefactos desde el Builder
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/server.js ./
COPY --from=builder /app/package.json ./

RUN chown -R citaplanner:nodejs /app

USER citaplanner

# Monitor de Salud del Nodo
HEALTHCHECK --interval=30s --timeout=10s --start-period=10s --retries=3 \
  CMD node -e "fetch('http://localhost:3000/api/settings/landing').then(r => r.ok ? process.exit(0) : process.exit(1)).catch(() => process.exit(1))"

EXPOSE 3000

# Activación del Nodo Maestro Aurum
CMD ["node", "server.js"]
