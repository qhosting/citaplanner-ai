# [AURUM PROTOCOL] - DOCKERFILE PRODUCTION
# PRODUCTO: CITAPLANNER ELITE BUSINESS SUITE
# REVISIÓN: 2026.4
# STATUS: PRODUCTION READY

# --- FASE 1: BUILDER ---
FROM node:20-alpine AS builder

# Instalación de dependencias del sistema para compilación
RUN apk add --no-cache libc6-compat

WORKDIR /app

# Copia de archivos de dependencias
COPY package*.json ./
# Instalamos todas las dependencias incluyendo devDeps para el build de Vite
RUN npm install

# Copia del código fuente
COPY . .

# Build del Frontend (Vite)
# Genera la carpeta /app/dist
RUN npm run build

# --- FASE 2: RUNNER ---
FROM node:20-alpine AS runner

WORKDIR /app

# Variables de entorno críticas
ENV NODE_ENV=production
ENV PORT=3000

# Seguridad: Creación de usuario no-root para ejecución segura
RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 citaplanner

# Directorio de persistencia para imágenes cargadas (Logos, Fotos de servicios)
RUN mkdir -p /app/uploads && chown -R citaplanner:nodejs /app/uploads

# Copia de artefactos de construcción desde la fase builder
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/server.js ./
COPY --from=builder /app/package*.json ./

# Instalación exclusiva de dependencias de producción para minimizar tamaño de imagen
RUN npm install --omit=dev

# Ajuste final de permisos para el usuario citaplanner en todo el directorio /app
RUN chown -R citaplanner:nodejs /app

# Cambio a usuario seguro
USER citaplanner

# Exposición del puerto del nodo Aurum (3000)
EXPOSE 3000

# Punto de entrada: Ejecución del servidor Node.js
# Se usa 'node server.js' asumiendo que package.json tiene "type": "module"
CMD ["node", "server.js"]

# NOTAS PARA DESPLIEGUE EN EASYPANEL:
# 1. Montar Volumen: /app/uploads (Tipo: Volume o Bind) para persistencia de archivos.
# 2. Variables Requeridas: 
#    - DATABASE_URL: URL de conexión a PostgreSQL.
#    - API_KEY: Clave de Google Gemini para las funciones de IA.
#    - NEXTAUTH_SECRET: Token de seguridad para cifrado de sesiones.
